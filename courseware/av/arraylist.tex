\begin{frame}
  \frametitle{ArrayList}
  \framesubtitle{Implementing a List using an array}
  \begin{block}{Theorem}
    \only<+->{An ArrayList implements the List interface.} 
    \only<+->{Ignoring the time
    spent in calls to $\mathsf{resize}()$,}
    \begin{itemize}
        \item<+->$\mathsf{get}(i)$ runs in $O(1)$ time;
        \item<+->$\mathsf{set}(i,x)$ run in $O(1)$ time;
        \item<+->$\mathsf{add}(i,x)$ runs in $O(1+n-i)$ time; and
	\item<+->$\mathsf{remove(i)}$ runs in $O(1+n-i)$ time.
    \end{itemize}
  \end{block}

  \begin{block}<+->{Question}
     What can we say about the cost of calls to $\mathsf{resize}()$?
  \end{block}
